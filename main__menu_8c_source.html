102void game_main_menu_on_init(void)
103{
104 affine_background_change_background(AFFINE_BG_MAIN_MENU);
105 change_background(BG_MAIN_MENU, false);
106 main_menu_ace = card_object_new(card_new(SPADES, ACE));
107 card_object_set_sprite(main_menu_ace, 0);
108 main_menu_ace->sprite_object->sprite->obj->attr0 |= ATTR0_AFF_DBL;
109 main_menu_ace->sprite_object->tx = int2fx(MAIN_MENU_ACE_T_X);
110 main_menu_ace->sprite_object->x = main_menu_ace->sprite_object->tx;
111 main_menu_ace->sprite_object->ty = int2fx(MAIN_MENU_ACE_T_Y);
112 main_menu_ace->sprite_object->y = main_menu_ace->sprite_object->ty;
113 main_menu_ace->sprite_object->tscale = float2fx(0.8f);
114 card_object_update(main_menu_ace);
115
116 // Select last highlighted button, Play button by default.
117 // e.g. if we return from the options menu, we want the Options button to be highlighted.
118 Selection sel_init = {last_highlighted_button, 0};
119 main_menu_selection_grid.selection = sel_init;
120
121 // Highlight current button
122 button_set_highlight(&main_menu_buttons[main_menu_selection_grid.selection.x], true);
123}

125void game_main_menu_on_update(void)
126{
127 card_object_update(main_menu_ace);
128 main_menu_ace->sprite_object->trotation = lu_sin((g_game_vars.timer << 8) / 2) / 3;
129 main_menu_ace->sprite_object->rotation = main_menu_ace->sprite_object->trotation;
130
131 // Seed randomization
132 g_game_vars.rng_seed++;
133 // If the keys have changed, make it more pseudo-random
134 if (key_curr_state() != key_prev_state())
135 {
136 g_game_vars.rng_seed *= 2;
137 }
138
139 selection_grid_process_input(&main_menu_selection_grid);
140}

178static void main_menu_on_key_transit(SelectionGrid* selection_grid, Selection* selection)
179{
180 // DEBUG: toggle git hash display by pressing L+R+START+SELECT
181 static bool combo_pressed = false;
182 static bool show_version = false;
183 if (key_is_down(KEY_START) && key_is_down(KEY_SELECT) && key_is_down(KEY_L) &&
184 key_is_down(KEY_R))
185 {
186 if (!combo_pressed)
187 {
188 show_version = !show_version;
189 if (show_version)
190 {
191 tte_printf(
192 "#{P:%d,%d; cx:0x%X000}%s",
193 GBALATRO_VERSION_TEXT_POS.x,
194 GBALATRO_VERSION_TEXT_POS.y,
195 TTE_WHITE_PB,
196 gbalatro_version
197 );
198 }
199 else
200 {
201 tte_erase_rect_wrapper(GBALATRO_VERSION_TEXT_RECT);
202 }
203 }
204
205 combo_pressed = true;
206 }
207 else
208 {
209 combo_pressed = false;
210 }
211
212 if (key_hit(SELECT_CARD))
213 {
214 button_press(&main_menu_buttons[selection->x]);
215 }
216}

64void game_main_menu_on_update()
65{
66 card_object_update(main_menu_ace);
67 main_menu_ace->sprite_object->trotation = lu_sin((g_game_vars.timer << 8) / 2) / 3;
68 main_menu_ace->sprite_object->rotation = main_menu_ace->sprite_object->trotation;
69
70 // Seed randomization
71 g_game_vars.rng_seed++;
72 // If the keys have changed, make it more pseudo-random
73 if (key_curr_state() != key_prev_state())
74 {
75 g_game_vars.rng_seed *= 2;
76 }
77
78 if (key_hit(KEY_LEFT))
79 {
80 if (selection_x > 0)
81 {
82 selection_x--;
83 }
84 }
85 else if (key_hit(KEY_RIGHT))
86 {
87 if (selection_x < IMPLEMENTED_BUTTONS - 1)
88 {
89 selection_x++;
90 }
91 }
92
93 if (selection_x == PLAY_BTN_IDX)
94 {
95 memset16(&pal_bg_mem[PLAY_BUTTON_OUTLINE_PID], BTN_HIGHLIGHT_COLOR, 1);
96
97 if (key_hit(SELECT_CARD))
98 {
99 play_sfx(SFX_BUTTON, MM_BASE_PITCH_RATE, BUTTON_SFX_VOLUME);
100 game_change_state(GAME_STATE_GAME_START);
101 }
102 }
103 else
104 {
105 memcpy16(&pal_bg_mem[PLAY_BUTTON_OUTLINE_PID], &pal_bg_mem[PLAY_BUTTON_MAIN_COLOR_PID], 1);
106 }
107}

958static inline void reroll_seed_str(void)
959{
960 // Need to roll a new seed and get a "real" random number. Otherwise the seed is still stuck at
961 // 0 and we'll have the same sequence of generated seeds each time.
962 // Also, don't use the shuffled seed as is, or we'll just end up with sequential seeds when
963 // rolling multiple times.
964 rng_shuffle_seed();
965 u32 new_seed = rng_get_u32();
966 rng_set_seed(new_seed);
967 u32_to_base36(new_seed, seed_str);
968 update_seed_text();
969 seed_cursor_pos = BASE36_MAX_DIGITS;
970}

975static inline void delete_seed_char(void)
976{
977 if (seed_cursor_pos == 0)
978 {
979 return;
980 }
981 seed_str[--seed_cursor_pos] = '\0';
982 update_seed_text();
983}

990static inline void type_seed_char(enum RunSetupKeyboardButtons key)
991{
992 if (seed_cursor_pos >= BASE36_MAX_DIGITS)
993 {
994 return;
995 }
996 seed_str[seed_cursor_pos++] = keyboard_buttons_to_char[key];
997 update_seed_text();
998}

1015static void keyboard_button_on_pressed(void)
1016{
1017 // Type something only if the button pressed in on the keyboard
1018 if (choose_seed_selection_grid.selection.y > RUN_SETUP_SEED_ROW_KEY3)
1019 return;
1020
1021 // Type only if the seed string is not full.
1022 // The cursor position is unsigned so always positive, but we still need to
1023 // ensure it doersn't go out of bounds by more than 1 so that we can always
1024 // substract 1 from it when erasing a character from the seed string.
1025 if (seed_cursor_pos > BASE36_MAX_DIGITS)
1026 {
1027 seed_cursor_pos = BASE36_MAX_DIGITS;
1028 }
1029
1030 if (key_hit(DESELECT_CARDS))
1031 {
1032 delete_seed_char();
1033 }
1034 else if (key_hit(SELECT_CARD))
1035 {
1036 // Get keyboard button index from selection
1037 enum RunSetupKeyboardButtons key =
1038 get_keyboard_index_from_sel(&choose_seed_selection_grid.selection);
1039
1040 switch (key)
1041 {
1042 case RUN_SETUP_KEYBOARD_RAND:
1043 reroll_seed_str();
1044 break;
1045 // Erase last character
1046 case RUN_SETUP_KEYBOARD_DEL:
1047 delete_seed_char();
1048 break;
1049 default:
1050 type_seed_char(key);
1051 break;
1052 }
1053 }
1054}

1290static void run_setup_tabs_update(void)
1291{
1292 static enum RunSetupTab current_tab = RUN_SETUP_TAB_RESUME;
1293
1294 // If there is no saved data, "Resume" tab is left grayed out and only one tab is available
1295 if (!is_saved_game_valid)
1296 {
1297 return;
1298 }
1299
1300 // Not all the way to the right and pressed R
1301 if (key_hit(TAB_RIGHT) && current_tab < RUN_SETUP_TAB_MAX - 1)
1302 {
1303 button_set_highlight(&tabs_buttons[current_tab], false);
1304 current_tab++;
1305 }
1306 // Not all the way to the left and pressed L
1307 else if (key_hit(TAB_LEFT) && current_tab > 0)
1308 {
1309 button_set_highlight(&tabs_buttons[current_tab], false);
1310 current_tab--;
1311 }
1312 // Either not pressed anything or no space to move, return early to not change button highlight
1313 else
1314 {
1315 return;
1316 }
1317
1318 button_set_highlight(&tabs_buttons[current_tab], true);
1319
1320 switch (current_tab)
1321 {
1322 case RUN_SETUP_TAB_NEW_RUN:
1323 state_machine_change_state(&run_setup_sm, RUN_SETUP_SUBSTATE_CHOOSE_DECK);
1324 break;
1325 case RUN_SETUP_TAB_RESUME:
1326 state_machine_change_state(&run_setup_sm, RUN_SETUP_SUBSTATE_RESUME);
1327 break;
1328 default:
1329 break;
1330 }
1331}

91// Fills the unbeaten boss blinds lists
92// This must be called at the beginning of a run
93void init_unbeaten_blinds_list(bool showdown)
94{
95 // create the lists when calling for the first time
96 static bool init = false;
97 if (!init)
98 {
99 init = true;
100 unbeaten_showdown_blinds = list_create();
101 unbeaten_boss_blinds = list_create();
102 }
103
104 List* p_unbeaten_blinds = showdown ? &unbeaten_showdown_blinds : &unbeaten_boss_blinds;
105
106 // empty the list just to be sure
107 list_clear(p_unbeaten_blinds);
108
109 int lower_blind = showdown ? BLIND_TYPE_SHOWDOWN : BLIND_TYPE_BOSS;
110 int upper_blind = showdown ? BLIND_TYPE_MAX - 1 : BLIND_TYPE_SHOWDOWN - 1;
111
112 for (int i = lower_blind; i <= upper_blind; i++)
113 {
114 list_push_back(p_unbeaten_blinds, &_blind_type_map[i]);
115 }
116}

118enum BlindType roll_blind_type(bool showdown)
119{
120 List* p_unbeaten_blinds = showdown ? &unbeaten_showdown_blinds : &unbeaten_boss_blinds;
121
122 // Fill the list with all blinds if it is empty
123 // (happens on startup or if we have beaten all blinds)
124 if (list_is_empty(p_unbeaten_blinds))
125 {
126 init_unbeaten_blinds_list(showdown);
127 }
128
129 // roll a random blind among the unbeaten ones
130 int random_blind_idx = rand() % list_get_len(p_unbeaten_blinds);
131 Blind* random_blind = list_get_at_idx(p_unbeaten_blinds, random_blind_idx);
132
133 return random_blind->type;
134}

136void set_blind_beaten(enum BlindType type)
137{
138 bool showdown = (type >= BLIND_TYPE_SHOWDOWN);
139 List* p_unbeaten_blinds = showdown ? &unbeaten_showdown_blinds : &unbeaten_boss_blinds;
140
141 // find the beaten blind idx in the list
142 int beaten_idx = 0;
143 Blind* beaten_blind = NULL;
144 ListItr itr = list_itr_create(p_unbeaten_blinds);
145
146 while ((beaten_blind = list_itr_next(&itr)))
147 {
148 if (beaten_blind->type == type)
149 {
150 break;
151 }
152 beaten_idx++;
153 }
154
155 if (beaten_idx < list_get_len(p_unbeaten_blinds))
156 {
157 list_remove_at_idx(p_unbeaten_blinds, beaten_idx);
158 }
159}

174u16 blind_get_color(enum BlindType type, enum BlindColorIndex index)
175{
176 // Do a little translation of palette idx -> custom array idx
177 // All blinds before the Mark are arranged in pairs with their palettte split in two
178 // | XX | 1 | 2 | 3 | 4 | 5 | 6 | 7 | -> first sprite
179 // | XX | 9 | 10 | 11 | 12 | 13 | 14 | 15 | -> second sprite
180 // so we need to offset the color indices by 8 for blinds with an odd type
181 // From the Mark onwards, all sprites are alone in their spritesheet, so no need to offset
182 u32 color_idx = index + ((type < BLIND_TYPE_MARK)
183 ? BLIND_TOKEN_PALETTE_SIZE * (type % BLIND_TOKENS_PER_SPRITESHEET)
184 : 0);
185 return blind_gfxPal[get_blind_spritesheet_idx(type)][color_idx];
186}

188void apply_blind_colors(enum BlindType type)
189{
190 // keep track of active boss blind spritesheet to copy colors only when changing
191 static u32 active_boss_spritesheet = BLIND_TYPE_MAX;
192 u32 new_spritesheet = get_blind_spritesheet_idx(type);
193
194 // No need to update normal blind palette
195 if (type < BLIND_TYPE_BOSS || active_boss_spritesheet == new_spritesheet)
196 {
197 return;
198 }
199
200 active_boss_spritesheet = new_spritesheet;
201
202 // Just copy the palette as is to the sprite palette bank
203 memcpy16(
204 &pal_obj_bank[BOSS_BLIND_PB],
205 blind_gfxPal[active_boss_spritesheet],
206 NUM_ELEM_IN_ARR(blind_gfx0Pal)
207 );
208}

216void apply_blind_tiles(enum BlindType type, int layer)
217{
218 u32 spritesheet_idx = get_blind_spritesheet_idx(type);

221 &tile_mem[TILE_MEM_OBJ_CHARBLOCK0_IDX][get_layer_tile_index(layer)],

223 BLIND_SPRITE_COPY_SIZE
224 );

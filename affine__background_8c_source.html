39void affine_background_update()
40{
41 if (REG_IE & IRQ_HBLANK) // High quality mode with HBLANK interrupt
42 {
43 s_affine_background_prep_bgaff_arr();
44 }
45 else // Low quality mode without HBLANK interrupt
46 {
47 _asx.scr_x = 0;
48 _asx.scr_y = 0;
49 _asx.tex_x += 5;
50 _asx.tex_y += 12;
51 // Scale the sine value to fit in a s16
52 _asx.sx = ((lu_sin(_timer * 100)) >> 8) + 256;
53 // Scale the sine value to fit in a s16
54 _asx.sy = ((lu_sin(_timer * 100 + 0x4000)) >> 8) + 256;
55 _asx.alpha = 0;
56
57 bg_rotscale_ex(&_bgaff_arr[0], &_asx);
58 REG_BG_AFFINE[AFFINE_BG_IDX] = _bgaff_arr[0];
59 }
60
61 _timer++;
62}

116static IWRAM_CODE void s_affine_background_prep_bgaff_arr()
117{
118 for (u16 vcount = 0; vcount < SCREEN_HEIGHT; vcount++)
119 {
120 const s32 timer_s32 = _timer << 8;
121 const s32 vcount_s32 = vcount << 8;
122 const s16 vcount_s16 = vcount;
123 const s32 vcount_sine = lu_sin(vcount_s32 + timer_s32 / ANIMATION_SPEED_DIVISOR);
124
125 _asx.scr_x = (SCREEN_WIDTH / 2);
126 // scr_y must equal vcount otherwise the background will have no vertical difference
127 _asx.scr_y = vcount_s16 - (SCREEN_HEIGHT / 2);
128 _asx.tex_x = (1000 * 1000) + (vcount_sine);
129 _asx.tex_y = (1000 * 1000);
130 _asx.sx = 128;
131 _asx.sy = 128;
132 _asx.alpha = vcount_sine + (timer_s32 / ANIMATION_SPEED_DIVISOR);
133
134 bg_rotscale_ex(&_bgaff_arr[vcount], &_asx);
135 }
136
137 /* HBLANK occurs after the scanline so REG_VCOUNT represents the
138 * the scanline that just passed, so when it's SCREEN_HEIGHT we will
139 * actually be updating the first line
140 */
141 _bgaff_arr[SCREEN_HEIGHT] = _bgaff_arr[0];
142}

26// Clips a rect of screenblock entries to a specified rect
27// The bounding rect is not required to be within screenblock boundaries
28static inline void clip_se_rect_to_bounding_rect(Rect* rect, const Rect* bounding_rect)
29{
30 rect->right = min(rect->right, bounding_rect->right);
31 rect->bottom = min(rect->bottom, bounding_rect->bottom);
32 rect->left = max(rect->left, bounding_rect->left);
33 rect->top = max(rect->top, bounding_rect->top);
34}

43// Clips a rect of screenblock entries to be within one step of
44// screenblock boundaries vertically depending on direction.
45static inline void clip_se_rect_within_step_of_full_screen_vert(
46 Rect* se_rect,
47 enum ScreenVertDir direction
48)
49{
50 Rect bounding_rect = FULL_SCREENBLOCK_RECT;
51 if (direction == SCREEN_UP)
52 {
53 bounding_rect.top += 1;
54 }
55 else if (direction == SCREEN_DOWN)
56 {
57 bounding_rect.bottom -= 1;
58 }
59
60 clip_se_rect_to_bounding_rect(se_rect, &bounding_rect);
61}

63// Internal static function to merge implementation of move/copy functions.
64static void bg_se_copy_or_move_rect_1_tile_vert(
65 u16 bg_sbb,
66 Rect se_rect,
67 enum ScreenVertDir direction,
68 bool move
69)
70{
71 if (se_rect.left > se_rect.right || (direction != SCREEN_UP && direction != SCREEN_DOWN))
72 {
73 return;
74 }
75
76 // Clip to avoid read/write overflow of the screenblock
77 clip_se_rect_within_step_of_full_screen_vert(&se_rect, direction);
78
79 int start = (direction == SCREEN_UP) ? se_rect.top : se_rect.bottom;
80 int end = (direction == SCREEN_UP) ? se_rect.bottom : se_rect.top;
81
82 for (int y = start; y != end - direction; y -= direction)
83 {
84 memcpy16(
85 &(se_mat[bg_sbb][y + direction][se_rect.left]),
86 &se_mat[bg_sbb][y][se_rect.left],
87 rect_width(&se_rect)
88 );
89 }
90
91 if (move)
92 {
93 memset16(&se_mat[bg_sbb][end][se_rect.left], 0x0000, rect_width(&se_rect));
94 }
95}

126void main_bg_se_copy_rect(Rect se_rect, BG_POINT dest_pos)
127{
128 if (se_rect.left > se_rect.right || se_rect.top > se_rect.bottom)
129 return;
130
131 // Clip to avoid screenblock overflow
132 clip_se_rect_to_screenblock(&se_rect);
133
134 int width = rect_width(&se_rect);
135 int height = rect_height(&se_rect);
136 SE tile_map[height][width];
137
138 // Copy the rect to the tile map
139 for (int sy = 0; sy < height; sy++)
140 {
141 memcpy16(&tile_map[sy][0], &se_mat[MAIN_BG_SBB][se_rect.top + sy][se_rect.left], width);
142 }
143
144 // TODO: Avoid overflow
145 // Copy the tilemap to the new rect position
146 for (int sy = 0; sy < height; sy++)
147 {
148 memcpy16(&se_mat[MAIN_BG_SBB][dest_pos.y + sy][dest_pos.x], &tile_map[sy][0], width);
149 }
150}

217static inline void main_bg_se_3w_copy_expand_left_right_sides(

219 const BG_POINT* src_left_pnt

235void main_bg_se_copy_expand_3x3_rect(Rect se_dest_rect, BG_POINT src_top_left_pnt)
236{
237 clip_se_rect_to_screenblock(&se_dest_rect);
238
239 int dest_rect_width = rect_width(&se_dest_rect);
240 int dest_rect_height = rect_height(&se_dest_rect);
241
242 // Verify the dest rect is at least 2x2
243 if (dest_rect_width < 2 || dest_rect_height < 2)
244 {
245 return;
246 }
247
248 // Copy the corners
249 main_bg_se_expand_3x3_copy_corners(
250 &se_dest_rect,
251 &src_top_left_pnt,
252 dest_rect_width,
253 dest_rect_height
254 );
255
256 // Copy top and bottom sides
257 main_bg_se_expand_3x3_copy_top_bottom(&se_dest_rect, &src_top_left_pnt, dest_rect_width);
258
259 BG_POINT src_middle_left_pnt = {src_top_left_pnt.x, src_top_left_pnt.y + 1};
260
261 // Avoid the corners when copying the sides
262 Rect dest_sides_rect = se_dest_rect;
263 dest_sides_rect.top += 1;
264 dest_sides_rect.bottom -= 1;
265
266 // Copy left and right sides
267 main_bg_se_3w_copy_expand_left_right_sides(&dest_sides_rect, &src_middle_left_pnt);
268
269 // Fill the center if needed
270 if (dest_rect_width > 2 && dest_rect_height > 2)
271 {
272 SE middle_fill_se = se_mat[MAIN_BG_SBB][src_top_left_pnt.y + 1][src_top_left_pnt.x + 1];
273 Rect dest_inner_fill_rect = {
274 se_dest_rect.left + 1,
275 se_dest_rect.top + 1,
276 se_dest_rect.right - 1,
277 se_dest_rect.bottom - 1
278 };
279 main_bg_se_fill_rect_with_se(middle_fill_se, dest_inner_fill_rect);
280 }
281}

312void update_text_rect_to_right_align_str(
313 Rect* rect,
314 const char* str,
315 enum OverflowDir overflow_direction
316)
317{
318 // TODO: Allow passing string length to avoid calling strlen()?
319 int str_len = strlen(str);
320 if (overflow_direction == OVERFLOW_LEFT)
321 {
322 rect->left = max(0, rect->right - str_len * TTE_CHAR_SIZE);
323 }
324 else if (overflow_direction == OVERFLOW_RIGHT)
325 {
326 int num_fitting_chars = rect_width(rect) / TTE_CHAR_SIZE;
327 if (str_len < num_fitting_chars)
328 rect->left += (num_fitting_chars - str_len) * TTE_CHAR_SIZE;
329 // else nothing is to be updated, entire rect is filled and may overflow
330 }
331}

333void update_text_rect_to_center_str(Rect* rect, const char* str, enum ScreenHorzDir bias_direction)
334{
335 if (rect == NULL || str == NULL)
336 {
337 return;
338 }
339
340 int text_width_chars = strlen(str);
341 int rect_width_chars = rect_width(rect) / TTE_CHAR_SIZE;
342
343 bool bias_right = (bias_direction == SCREEN_RIGHT);
344
345 /* Adding bias_right makes sure that we round up when biased right
346 * but round down when biased left.
347 */
348 rect->left += max(0, (rect_width_chars - text_width_chars + bias_right) / 2) * TTE_CHAR_SIZE;
349}

351void memcpy16_tile8_with_palette_offset(u16* dst, const u16* src, uint hwcount, u8 palette_offset)
352{
353 const u16 offset = (((palette_offset) << 8) | (palette_offset));
354 for (int i = 0; i < hwcount; i++)
355 {
356 // Copying u8 data twice across u16 data
357 dst[i] = src[i] + offset;

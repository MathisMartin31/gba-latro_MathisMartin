142bool list_swap(List* list, unsigned int idx_a, unsigned int idx_b)
143{
144 if (idx_a >= list->len || idx_b >= list->len)
145 return false;
146 if (idx_a == idx_b)
147 return true; // swapping with yourself isn't technically an error
148
149 unsigned int curr_idx = 0;
150 unsigned int max_idx = idx_a > idx_b ? idx_a : idx_b;
151 ListNode* node_a = NULL;
152 ListNode* node_b = NULL;
153
154 ListItr itr = list_itr_create(list);
155 ListNode* ln;
156
157 do
158 {
159 ln = s_list_itr_node_next(&itr);
160 if (idx_a == curr_idx)
161 {
162 node_a = ln;
163 continue;
164 }
165 if (idx_b == curr_idx)
166 {
167 node_b = ln;
168 continue;
169 }
170 } while (max_idx != curr_idx++);
171
172 // Just swap the data pointers
173 void* tmp = node_a->data;
174 node_a->data = node_b->data;
175 node_b->data = tmp;
176
177 return true;
178}

180static void s_list_remove_node(List* list, ListNode* node)
181{
182 if (node->prev && !node->next) // end of list
183 {
184 node->prev->next = NULL;
185 list->tail = node->prev;
186 }
187 else if (node->prev && node->next) // somewhere in between
188 {
189 node->prev->next = node->next;
190 node->next->prev = node->prev;
191 }
192 else if (node->next && !node->prev) // beginning of list
193 {
194 node->next->prev = NULL;
195 list->head = node->next;
196 }
197 else if (!node->prev && !node->next) // only element in list
198 {
199 list->head = NULL;
200 list->tail = NULL;
201 }
202
203 POOL_FREE(ListNode, node);
204
205 list->len--;
206}

281static ListNode* s_list_itr_node_next(ListItr* itr)
282{
283 if (!itr->next_node)
284 return NULL;
285
286 itr->current_node = itr->next_node;
287
288 ListNode* ln = itr->next_node;
289 ListNode* next_itr_node = (itr->direction == LIST_ITR_FORWARD) ? ln->next : ln->prev;
290
291 if (next_itr_node)
292 {
293 itr->next_node = next_itr_node;
294 return ln;
295 }
296
297 itr->next_node = NULL;
298 return ln;

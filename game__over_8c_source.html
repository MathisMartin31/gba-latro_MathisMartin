40static void game_over_init(enum EndCondition init_condition)
41{
42 condition = init_condition;
43 // Clears the round end menu
44 main_bg_se_clear_rect(POP_MENU_ANIM_RECT);
45 main_bg_se_copy_expand_3x3_rect(GAME_OVER_DIALOG_DEST_RECT, GAME_OVER_SRC_RECT_3X3_POS);
46 main_bg_se_copy_rect(NEW_RUN_BTN_SRC_RECT, NEW_RUN_BTN_DEST_POS);
47
48 // Using the text color to match the Win/Loss condition
49 switch (condition)
50 {
51 case END_CONDITION_WIN:
52 affine_background_set_color(TEXT_CLR_BLUE);
53 break;
54 case END_CONDITION_LOSS:
55 affine_background_set_color(TEXT_CLR_RED);
56 break;
57 default:
58 break;
59 }
60
61 timer = TM_ZERO;
62}

94void game_over_on_update(void)
95{
96 timer++;
97 if (timer < GAME_OVER_ANIM_FRAMES)
98 {
99 game_over_anim_frame();
100 }
101 else if (timer == GAME_OVER_ANIM_FRAMES)
102 {
103 switch (condition)
104 {
105 case END_CONDITION_WIN:
106 tte_printf(
107 "#{P:%d,%d; cx:0x%X000}YOU WIN",
108 GAME_WIN_MSG_TEXT_RECT.left,
109 GAME_WIN_MSG_TEXT_RECT.top,
110 TTE_BLUE_PB
111 );
112 break;
113 case END_CONDITION_LOSS:
114 tte_printf(
115 "#{P:%d,%d; cx:0x%X000}GAME OVER",
116 GAME_LOSE_MSG_TEXT_RECT.left,
117 GAME_LOSE_MSG_TEXT_RECT.top,
118 TTE_RED_PB
119 );
120 break;
121 default:
122 break;
123 }
124 }
125 game_over_process_user_input();
126}

219static inline int game_shop_get_rand_available_joker_id(void)
220{
221 // Roll for what rarity the joker will be
222 int joker_rarity = joker_get_random_rarity();
223
224 // Now determine how many jokers are available based on the rarity
225 int jokers_avail_size = get_num_shop_jokers_avail();
226
227 if (jokers_avail_size == 0)
228 return UNDEFINED;
229
230 int matching_joker_ids[jokers_avail_size];
231 int fallback_random_idx = rng_get_u32() % jokers_avail_size;
232 int fallback_random_joker_id = UNDEFINED;
233 int match_count = 0;
234
235 BitsetItr itr = bitset_itr_create(&s_avail_jokers_bitset);
236
237 int i = 0;
238 int joker_id = UNDEFINED;
239 while ((joker_id = bitset_itr_next(&itr)) != UNDEFINED)
240 {
241 if (i++ == fallback_random_idx)
242 fallback_random_joker_id = joker_id;
243 const JokerInfo* info = get_joker_registry_entry(joker_id);
244 if (info->rarity == joker_rarity)
245 {
246 matching_joker_ids[match_count++] = joker_id;
247 }
248 }
249
250 int selected_joker_id = (match_count > 0) ? matching_joker_ids[rng_get_u32() % match_count]
251 : fallback_random_joker_id;
252
253 return selected_joker_id;
254}

275static void game_shop_create_items(void)
276{
277 tte_erase_rect_wrapper(SHOP_PRICES_TEXT_RECT);
278
279 if (no_avail_jokers())
280 return;
281
282 List* shop_jokers_list = &s_shop_jokers_list;
283
284 list_clear(shop_jokers_list);
285 *shop_jokers_list = list_init();
286
287 for (int i = 0; i < MAX_SHOP_JOKERS; i++)
288 {
289 int joker_id = 0;
290#ifdef TEST_JOKER_ID0 // Allow defining an ID for a joker to always appear in shop and be tested
291 if (is_shop_joker_avail(TEST_JOKER_ID0))
292 {
293 joker_id = TEST_JOKER_ID0;
294 }
295 else
296#endif
297#ifdef TEST_JOKER_ID1
298 if (is_shop_joker_avail(TEST_JOKER_ID1))
299 {
300 joker_id = TEST_JOKER_ID1;
301 }
302 else
303#endif
304 {
305 joker_id = game_shop_get_rand_available_joker_id();
306 }
307
308 // If for some reason only no joker is left, don't make another
309 if (joker_id == UNDEFINED)
310 break;
311
312 game_shop_set_joker_avail(joker_id, false);
313
314 JokerObject* joker_object = joker_object_new(joker_new(joker_id));
315
316 joker_object->sprite_object->x =
317 int2fx(SHOP_JOKER_SPRITES_INIT_POS.x + i * CARD_SPRITE_SIZE);
318 joker_object->sprite_object->y = int2fx(SHOP_JOKER_SPRITES_INIT_POS.y);
319 joker_object->sprite_object->tx = joker_object->sprite_object->x;
320 joker_object->sprite_object->ty = int2fx(ITEM_SHOP_Y);
321
322 sprite_object_print_price_under(joker_object->sprite_object, joker_object->joker->value);
323
324 sprite_position(
325 joker_object_get_sprite(joker_object),
326 fx2int(joker_object->sprite_object->x),
327 fx2int(joker_object->sprite_object->y)
328 );
329
330 list_push_back(shop_jokers_list, joker_object);
331 }

446static bool shop_top_row_on_selection_changed(
447 SelectionGrid* selection_grid,
448 int row_idx,
449 const Selection* prev_selection,
450 const Selection* new_selection
451)
452{
453 List* shop_jokers_list = &s_shop_jokers_list;
454 // Guard if we move down while on jokers
455 if (new_selection->y > row_idx && prev_selection->x > 0)
456 return false;
457
458 // The selection grid system only guarantees that the new selection is within bounds
459 // but not the previous one...
460 // This allows using INIT_SEL = {-1, 1} and move to set the initial selection in a hacky way...
461 if (prev_selection->y == row_idx && prev_selection->x >= 0 &&
462 prev_selection->x < shop_top_row_get_size())
463 {
464 if (prev_selection->x == NEXT_ROUND_BTN_SEL_X)
465 {
466 button_set_highlight(&next_round_button, false);
467 }
468 else
469 {
470 int idx = prev_selection->x - 1; // -1 to account for next round button
471 JokerObject* joker_object = (JokerObject*)list_get_at_idx(shop_jokers_list, idx);
472 sprite_object_set_focus(joker_object->sprite_object, false);
473 }
474 }
475
476 if (new_selection->y == row_idx)
477 {
478 if (new_selection->x == NEXT_ROUND_BTN_SEL_X)
479 {
480 button_set_highlight(&next_round_button, true);
481 }
482 else
483 {
484 int idx = new_selection->x - 1; // -1 to account for next round button
485 JokerObject* joker_object = (JokerObject*)list_get_at_idx(shop_jokers_list, idx);
486 sprite_object_set_focus(joker_object->sprite_object, true);
487 }
488 }
489
490 return true;
491}

536static inline void game_shop_reroll(int* reroll_cost)
537{
538 g_game_vars.money -= *reroll_cost;
539 display_money(); // Update the money display
540
541 List* shop_jokers_list = &s_shop_jokers_list;
542 ListItr itr = list_itr_create(shop_jokers_list);
543 JokerObject* joker_object;
544
545 while ((joker_object = list_itr_next(&itr)))
546 {
547 if (joker_object != NULL)
548 {
549 game_shop_set_joker_avail(joker_object->joker->id, true);
550 joker_object_destroy(&joker_object); // Destroy the joker object if it exists
551 }
552 }
553
554 list_clear(shop_jokers_list);
555 *shop_jokers_list = list_init();
556
557 game_shop_create_items();
558
559 itr = list_itr_create(shop_jokers_list);
560
561 while ((joker_object = list_itr_next(&itr)))
562 {
563 if (joker_object != NULL)
564 {
565 // Set the y position to the target position
566 joker_object->sprite_object->y = joker_object->sprite_object->ty;
567
568 // Give the joker a little wiggle animation
569 joker_object_shake(joker_object, UNDEFINED);
570 }
571 }
572
573 (*reroll_cost)++;
574 tte_printf(
575 "#{P:%d,%d; cx:0x%X000}$%d",
576 SHOP_REROLL_RECT.left,
577 SHOP_REROLL_RECT.top,
578 TTE_WHITE_PB,
579 *reroll_cost
580 );
581}

671static inline void game_shop_lights_anim_frame(void)
672{
673 // Shift palette around the border of the shop icon
674 COLOR shifted_palette[4];
675 shifted_palette[0] = pal_bg_mem[SHOP_LIGHTS_2_PAL_IDX];
676 shifted_palette[1] = pal_bg_mem[SHOP_LIGHTS_3_PAL_IDX];
677 shifted_palette[2] = pal_bg_mem[SHOP_LIGHTS_4_PAL_IDX];
678 shifted_palette[3] = pal_bg_mem[SHOP_LIGHTS_1_PAL_IDX];
679
680 // Circularly shift the palette
681 int last = shifted_palette[3];
682
683 for (int i = 3; i > 0; --i)
684 {
685 shifted_palette[i] = shifted_palette[i - 1];
686 }
687
688 shifted_palette[0] = last;
689
690 // Copy the shifted palette to the next 4 slots
691 pal_bg_mem[SHOP_LIGHTS_2_PAL_IDX] = shifted_palette[0];
692 pal_bg_mem[SHOP_LIGHTS_3_PAL_IDX] = shifted_palette[1];
693 pal_bg_mem[SHOP_LIGHTS_4_PAL_IDX] = shifted_palette[2];
694 pal_bg_mem[SHOP_LIGHTS_1_PAL_IDX] = shifted_palette[3];
695}

260static inline int game_shop_get_rand_available_joker_id(void)
261{
262 // Roll for what rarity the joker will be
263 int joker_rarity = joker_get_random_rarity();
264
265 // Now determine how many jokers are available based on the rarity
266 int jokers_avail_size = get_num_shop_jokers_avail();
267
268 if (jokers_avail_size == 0)
269 return UNDEFINED;
270
271 int matching_joker_ids[jokers_avail_size];
272 int fallback_random_idx = rng_get_u32() % jokers_avail_size;
273 int fallback_random_joker_id = UNDEFINED;
274 int match_count = 0;
275
276 BitsetItr itr = bitset_itr_create(&s_avail_jokers_bitset);
277
278 int i = 0;
279 int joker_id = UNDEFINED;
280 while ((joker_id = bitset_itr_next(&itr)) != UNDEFINED)
281 {
282 if (i++ == fallback_random_idx)
283 fallback_random_joker_id = joker_id;
284 const JokerInfo* info = get_joker_registry_entry(joker_id);
285 if (info->rarity == joker_rarity)
286 {
287 matching_joker_ids[match_count++] = joker_id;
288 }
289 }
290
291 int selected_joker_id = (match_count > 0) ? matching_joker_ids[rng_get_u32() % match_count]
292 : fallback_random_joker_id;
293
294 return selected_joker_id;
295}

316static void game_shop_create_items(void)
317{
318 tte_erase_rect_wrapper(SHOP_PRICES_TEXT_RECT);
319
320 if (no_avail_jokers())
321 return;
322
323 List* shop_jokers_list = &s_shop_jokers_list;
324
325 list_clear(shop_jokers_list);
326 *shop_jokers_list = list_init();
327
328 for (int i = 0; i < MAX_SHOP_JOKERS; i++)
329 {
330 int joker_id = 0;
331#ifdef TEST_JOKER_ID0 // Allow defining an ID for a joker to always appear in shop and be tested
332 if (is_shop_joker_avail(TEST_JOKER_ID0))
333 {
334 joker_id = TEST_JOKER_ID0;
335 }
336 else
337#endif
338#ifdef TEST_JOKER_ID1
339 if (is_shop_joker_avail(TEST_JOKER_ID1))
340 {
341 joker_id = TEST_JOKER_ID1;
342 }
343 else
344#endif
345 {
346 joker_id = game_shop_get_rand_available_joker_id();
347 }
348
349 // If for some reason only no joker is left, don't make another
350 if (joker_id == UNDEFINED)
351 break;
352
353 game_shop_set_joker_avail(joker_id, false);
354
355 JokerObject* joker_object = joker_object_new(joker_new(joker_id));
356
357 joker_object->sprite_object->x =
358 int2fx(SHOP_JOKER_SPRITES_INIT_POS.x + i * CARD_SPRITE_SIZE);
359 joker_object->sprite_object->y = int2fx(SHOP_JOKER_SPRITES_INIT_POS.y);
360 joker_object->sprite_object->tx = joker_object->sprite_object->x;
361 joker_object->sprite_object->ty = int2fx(ITEM_SHOP_Y);
362
363 sprite_object_print_price_under(joker_object->sprite_object, joker_object->joker->value);
364
365 list_push_back(shop_jokers_list, joker_object);
366 }

481static bool shop_top_row_on_selection_changed(
482 SelectionGrid* selection_grid,
483 int row_idx,
484 const Selection* prev_selection,
485 const Selection* new_selection
486)
487{
488 List* shop_jokers_list = &s_shop_jokers_list;
489 // Guard if we move down while on jokers
490 if (new_selection->y > row_idx && prev_selection->x > 0)
491 return false;
492
493 // The selection grid system only guarantees that the new selection is within bounds
494 // but not the previous one...
495 // This allows using INIT_SEL = {-1, 1} and move to set the initial selection in a hacky way...
496 if (prev_selection->y == row_idx && prev_selection->x >= 0 &&
497 prev_selection->x < shop_top_row_get_size())
498 {
499 if (prev_selection->x == NEXT_ROUND_BTN_SEL_X)
500 {
501 button_set_highlight(&next_round_button, false);
502 }
503 else
504 {
505 int idx = prev_selection->x - 1; // -1 to account for next round button
506 JokerObject* joker_object = (JokerObject*)list_get_at_idx(shop_jokers_list, idx);
507 sprite_object_set_focus(joker_object->sprite_object, false);
508 }
509 }
510
511 if (new_selection->y == row_idx)
512 {
513 if (new_selection->x == NEXT_ROUND_BTN_SEL_X)
514 {
515 button_set_highlight(&next_round_button, true);
516 }
517 else
518 {
519 int idx = new_selection->x - 1; // -1 to account for next round button
520 JokerObject* joker_object = (JokerObject*)list_get_at_idx(shop_jokers_list, idx);
521 sprite_object_set_focus(joker_object->sprite_object, true);
522 }
523 }
524
525 return true;
526}

571static inline void game_shop_reroll(void)
572{
573 g_game_vars.money -= reroll_cost;
574 display_money(); // Update the money display
575
576 List* shop_jokers_list = &s_shop_jokers_list;
577 ListItr itr = list_itr_create(shop_jokers_list);
578 JokerObject* joker_object;
579
580 while ((joker_object = list_itr_next(&itr)))
581 {
582 if (joker_object != NULL)
583 {
584 game_shop_set_joker_avail(joker_object->joker->id, true);
585 joker_object_destroy(&joker_object); // Destroy the joker object if it exists
586 }
587 }
588
589 list_clear(shop_jokers_list);
590 *shop_jokers_list = list_init();
591
592 game_shop_create_items();
593
594 itr = list_itr_create(shop_jokers_list);
595
596 while ((joker_object = list_itr_next(&itr)))
597 {
598 if (joker_object != NULL)
599 {
600 // Set the y position to the target position
601 joker_object->sprite_object->y = joker_object->sprite_object->ty;
602
603 // Give the joker a little wiggle animation
604 joker_object_shake(joker_object, UNDEFINED);
605 }
606 }
607
608 reroll_cost++;
609 tte_printf(
610 "#{P:%d,%d; cx:0x%X000}$%d",
611 SHOP_REROLL_RECT.left,
612 SHOP_REROLL_RECT.top,
613 TTE_WHITE_PB,
614 reroll_cost
615 );
616}

655static void game_shop_process_user_input(void)
656{
657 selection_grid_process_input(&shop_selection_grid);
658
659 static JokerObject* tmp_card = NULL;
660
661 // Determine the Joker we would show the description of
662 switch (shop_selection_grid.selection.y)
663 {
664 // Owned Joker
665 case 0:
666 {
667 description_card_original_list = get_jokers_list();
668 tmp_card = list_get_at_idx(get_jokers_list(), shop_selection_grid.selection.x);
669 break;
670 }
671
672 // Jokers for sale
673 case 1:
674 {
675 description_card_original_list = &s_shop_jokers_list;
676 tmp_card = (shop_selection_grid.selection.x > 0)
677 ? list_get_at_idx(&s_shop_jokers_list, shop_selection_grid.selection.x - 1)
678 : NULL;
679 break;
680 }
681
682 // TODO: handle Consumables and Vouchers when implemented
683
684 default:
685 {
686 description_card_original_list = NULL;
687 tmp_card = NULL;
688 break;
689 }
690 }
691
692 // Show description of selected card when pressing B.
693 // Always wait for the card in question to be immobile to avoid accumulating
694 // errors when pressing and releasing B in quick succession.
695 if (tmp_card != NULL && tmp_card->sprite_object->vx == 0 && tmp_card->sprite_object->vy == 0 &&
696 key_held(DESELECT_CARDS))
697 {
698 description_card = tmp_card;

963static inline void game_shop_lights_anim_frame(void)
964{
965 // Shift palette around the border of the shop icon
966 COLOR shifted_palette[4];
967 shifted_palette[0] = pal_bg_mem[SHOP_LIGHTS_2_PAL_IDX];
968 shifted_palette[1] = pal_bg_mem[SHOP_LIGHTS_3_PAL_IDX];
969 shifted_palette[2] = pal_bg_mem[SHOP_LIGHTS_4_PAL_IDX];
970 shifted_palette[3] = pal_bg_mem[SHOP_LIGHTS_1_PAL_IDX];
971
972 // Circularly shift the palette
973 int last = shifted_palette[3];
974
975 for (int i = 3; i > 0; --i)
976 {
977 shifted_palette[i] = shifted_palette[i - 1];
978 }
979
980 shifted_palette[0] = last;
981
982 // Copy the shifted palette to the next 4 slots
983 pal_bg_mem[SHOP_LIGHTS_2_PAL_IDX] = shifted_palette[0];
984 pal_bg_mem[SHOP_LIGHTS_3_PAL_IDX] = shifted_palette[1];
985 pal_bg_mem[SHOP_LIGHTS_4_PAL_IDX] = shifted_palette[2];
986 pal_bg_mem[SHOP_LIGHTS_1_PAL_IDX] = shifted_palette[3];
987}

19// Damping Constants: SPRING_DAMP_NUMERATOR/2^SPRING_DAMP_DENOM_SHIFT ~ 0.699 damping factor
20//
21// SPRING_DAMP_ROUNDING = 2^(SPRING_DAMP_DENOM_SHIFT - 1) rounding for fixed-point arithmetic by
22// adding half the denominator to round instead of truncating
23#define SPRING_DAMP_NUMERATOR 179
24#define SPRING_DAMP_DENOM_SHIFT 8
25#define SPRING_DAMP_ROUNDING (1 << (SPRING_DAMP_DENOM_SHIFT - 1))

36Sprite* sprite_new(u16 a0, u16 a1, u32 tid, u32 pb, int sprite_index)
37{
38 Sprite* sprite = POOL_GET(Sprite);
39
40 sprite->obj = NULL;
41 sprite->aff = NULL;
42
43 if (!free_sprites[sprite_index])
44 {
45 free_sprites[sprite_index] = sprite;
46 }
47 else
48 {
49 POOL_FREE(Sprite, sprite);
50 return NULL;
51 }
52
53 if (a0 & ATTR0_AFF)
54 {
55 int aff_index = MAX_AFFINES;
56
57 for (int i = 0; i < MAX_AFFINES; i++)
58 {
59 if (!free_affines[i])
60 {
61 free_affines[i] = true;
62 aff_index = i;
63 break;
64 }
65 }
66
67 if (aff_index == MAX_AFFINES)
68 {
69 POOL_FREE(Sprite, sprite);
70 return NULL;
71 }
72
73 a1 = a1 | ATTR1_AFF_ID(aff_index);
74
75 sprite->obj = &obj_buffer[sprite_index];
76 sprite->aff = &obj_aff_buffer[aff_index];
77 obj_set_attr(sprite->obj, a0, a1, ATTR2_PALBANK(pb) | tid);
78 obj_aff_identity(&obj_aff_buffer[aff_index]);
79 }
80 else
81 {
82 sprite->obj = &obj_buffer[sprite_index];
83 obj_set_attr(sprite->obj, a0, a1, ATTR2_PALBANK(pb) | tid);
84 }
85
86 sprite->idx = sprite_index;
87
88 return sprite;
89}

231static inline IWRAM_CODE void update_sprite_position(SpriteObject* sprite_object)
232{
233 sprite_object->vx += ((sprite_object->tx - sprite_object->x) * g_game_vars.game_speed) / 8;
234 sprite_object->vy += ((sprite_object->ty - sprite_object->y) * g_game_vars.game_speed) / 8;
235
236 // Scale up the card when it's played
237 sprite_object->vscale += (sprite_object->tscale - sprite_object->scale) / 8;
238
239 // Rotate the card when it's played
240 sprite_object->vrotation += (sprite_object->trotation - sprite_object->rotation) / 8;
241
242 const FIXED epsilon = (FIX_ONE >> 6); // = 1/2^6 = 0.015625
243
244 // Snap to target position when velocity is negligible to avoid infinite approach
245 if (abs(sprite_object->vx) < epsilon && abs(sprite_object->vy) < epsilon)
246 {
247 sprite_object->vx = 0;
248 sprite_object->vy = 0;
249
250 sprite_object->x = sprite_object->tx;
251 sprite_object->y = sprite_object->ty;
252 }
253 else
254 {
255 sprite_object->vx = (sprite_object->vx * SPRING_DAMP_NUMERATOR + SPRING_DAMP_ROUNDING) >>
256 SPRING_DAMP_DENOM_SHIFT;
257 sprite_object->vy = (sprite_object->vy * SPRING_DAMP_NUMERATOR + SPRING_DAMP_ROUNDING) >>
258 SPRING_DAMP_DENOM_SHIFT;
259
260 sprite_object->x += sprite_object->vx;
261 sprite_object->y += sprite_object->vy;
262 }
263
264 // Set scale to 0 if it's close enough to the target
265 if (abs(sprite_object->vscale) < epsilon)
266 {
267 sprite_object->vscale = 0;
268 sprite_object->scale = sprite_object->tscale;
269 }
270 else
271 {
272 sprite_object->vscale =
273 (sprite_object->vscale * SPRING_DAMP_NUMERATOR + SPRING_DAMP_ROUNDING) >>
274 SPRING_DAMP_DENOM_SHIFT;
275 sprite_object->scale += sprite_object->vscale;
276 }
277
278 // For rotation, prioritize snapping to target if close enough, then zero velocity.
279 if (abs(sprite_object->vrotation) < epsilon)
280 {
281 sprite_object->vrotation = 0;
282 sprite_object->rotation = sprite_object->trotation;
283 }
284 else // Apply damping and update rotation if not yet settled
285 {
286 sprite_object->vrotation =
287 (sprite_object->vrotation * SPRING_DAMP_NUMERATOR + SPRING_DAMP_ROUNDING) >>
288 SPRING_DAMP_DENOM_SHIFT;
289 sprite_object->rotation += sprite_object->vrotation;
290 }
291
292 // Apply rotation and scale to the sprite
293 obj_aff_rotscale(
294 sprite_object->sprite->aff,
295 sprite_object->scale,
296 sprite_object->scale,
297 -sprite_object->vx + sprite_object->rotation
298 );
299}

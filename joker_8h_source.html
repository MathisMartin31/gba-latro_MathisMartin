48// These are the common Joker Events. Special Joker behaviour will be checked on a
49// Joker per Joker basis (see if it's there, then do something, e.g. Pareidolia, Baseball Card)
50enum JokerEvent
51{
52 JOKER_EVENT_ON_JOKER_CREATED, // Triggers only once when the Joker is created, mainly used for
53 // data initialization
54 JOKER_EVENT_ON_HAND_PLAYED, // Triggers only once when the hand is played and before the cards
55 // are scored
56 JOKER_EVENT_ON_CARD_SCORED, // Triggers when a played card scores (e.g. Walkie Talkie,
57 // Fibonnacci...)
58 JOKER_EVENT_ON_CARD_SCORED_END, // Triggers after the card has finishd scoring (e.g. retrigger
59 // Jokers)
60 JOKER_EVENT_ON_CARD_HELD, // Triggers when going through held cards
61 JOKER_EVENT_INDEPENDENT, // Joker will trigger normally, when Jokers are scored (e.g. base
62 // Joker)
63 JOKER_EVENT_ON_HAND_SCORED_END, // Triggers when entire hand has finished scoring (e.g. food
64 // Jokers)
65 JOKER_EVENT_ON_HAND_DISCARDED, // Triggers when discarding a hand
66 JOKER_EVENT_ON_ROUND_END, // Triggers at the end of the round (e.g. Rocket)
67 JOKER_EVENT_ON_BLIND_SELECTED, // Triggers when selecting a blind (e.g. Dagger, Riff Raff,
68 // Madness..)
69};

55// These are the common Joker Events. Special Joker behaviour will be checked on a
56// Joker per Joker basis (see if it's there, then do something, e.g. Pareidolia, Baseball Card)
57enum JokerEvent
58{
59 JOKER_EVENT_ON_JOKER_CREATED, // Triggers only once when the Joker is created, mainly used for
60 // data initialization
61 JOKER_EVENT_ON_HAND_PLAYED, // Triggers only once when the hand is played and before the cards
62 // are scored
63 JOKER_EVENT_ON_CARD_SCORED, // Triggers when a played card scores (e.g. Walkie Talkie,
64 // Fibonnacci...)
65 JOKER_EVENT_ON_CARD_SCORED_END, // Triggers after the card has finishd scoring (e.g. retrigger
66 // Jokers)
67 JOKER_EVENT_ON_CARD_HELD, // Triggers when going through held cards
68 JOKER_EVENT_INDEPENDENT, // Joker will trigger normally, when Jokers are scored (e.g. base
69 // Joker)
70 JOKER_EVENT_ON_HAND_SCORED_END, // Triggers when entire hand has finished scoring (e.g. food
71 // Jokers)
72 JOKER_EVENT_ON_HAND_DISCARDED, // Triggers when discarding a hand
73 JOKER_EVENT_ON_ROUND_END, // Triggers at the end of the round (e.g. Rocket)
74 JOKER_EVENT_ON_BLIND_SELECTED, // Triggers when selecting a blind (e.g. Dagger, Riff Raff,
75 // Madness..)
76};

256static inline bool get_save_header(SaveHeader* header)
257{
258 read_sram(HEADER_ADDRESS, (u8*)header, sizeof(*header));
259 return (header->magic == CHECK_MAGIC) && header->dirty == is_version_dirty() &&
260 check_hash(header->githash);
261}

274static inline void set_save_header(u32 section_flag)
275{
276 SaveHeader header;
277
278 // Check for valid data. If it's junk, set all sections as invalid, else keep the flags.
279 // Then add the requested flag.
280 if (!get_save_header(&header))
281 {
282 memcpy(&header, &SaveHeader_default, sizeof(SaveHeader_default));
283 }
284
285 header.valid_sections |= section_flag;
286
287 header.dirty = is_version_dirty();
288 memcpy(&(header.githash), gbalatro_version + GIT_HASH_START, CHECK_HASH_SIZE);
289
290 write_sram(HEADER_ADDRESS, (const u8*)&header, sizeof(header));
291}

352 (u32)joker_object->joker->id,
353 joker_object->joker->persistent_state
354 };
355 game.jokers_data[i] = data;
356 }
357 for (; i < MAX_JOKERS_HELD_SIZE; i++)
358 {
359 JokerObjectSaveData data = {UNDEFINED, UNDEFINED};
360 game.jokers_data[i] = data;
361 }
362
363 write_sram(GAME_ADDRESS, (const u8*)&game, sizeof(game));
364 set_save_header(SAVE_SECTION_FLAG_GAME);
365}
